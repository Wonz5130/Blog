### 五、算法设计题（1*10'=10'）

求任意顶点 v 的入度。(这题还好，上午刚写过这个算法)

```C++
typedef struct ENode{
    int adjVex;
    struct ENode *nextArc;
}ENode;
 
typedef struct{
    int n;
    ENode **A;
}LGraph;
 
int Degree(LGraph g,int v){
    int i,d = 0;
    ENode *p;
    for(i = 0;i < g.n;i ++){
        for(p = g.A[i];p;p = p->nextArc){
            if(p->adjVex == v){
                d ++;
            }
        }
    }
    return d;
}
```
